class Carro {
  int portas = 0;
  int rodas = 4;
  int velocidade = 0;
  Carro(this.portas);
  String estado() {
    return 'Carro com $portas portas. Velocidade: $velocidade km/h';
  }
  int acelera() => velocidade = velocidade + 5;
  
  int freia() => (velocidade <= 10) ? velocidade = 0 : velocidade = velocidade - 10;
}

void main() {
  Carro novo = Carro(4);
  print(novo.estado());
  novo.acelera();
  novo.acelera();
  novo.acelera();
  print(novo.estado());
  novo.freia();
  print(novo.estado());
  novo.freia();
  print(novo.estado());
}
