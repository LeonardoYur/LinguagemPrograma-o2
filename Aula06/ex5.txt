import 'dart:io';

Future<void> mostraHoras() async {
  var arquivo = File('horas.txt');

  if (await arquivo.exists()) {
    String conteudo = await arquivo.readAsString();
    print(conteudo);
  } else {
    print('Arquivo não encontrado');
  }
}

void main() async {
  await mostraHoras();
}
