import 'dart:io';

void main(List<String> arguments) async {
  print('Content-Type:text/html;charset=iso-8859-1;\n');

  var arquivo = File('churrasco.txt');

  print('<html>');
  print('<body>');
  print('<h1>Lista do Churrasco</h1>');

  if (await arquivo.exists()) {
    String conteudo = await arquivo.readAsString();

    List<String> nomes = conteudo.split('\n');

    print('<ul>');

    for (var nome in nomes) {
      if (nome.trim().isNotEmpty) {
        print('<li>$nome</li>');
      }
    }

    print('</ul>');
  } else {
    print('<p>Nenhum nome cadastrado.</p>');
  }

  print('</body>');
  print('</html>');
}
