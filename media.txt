void main() {
  var notas = [4.5, 9.2, 7.8];
  var mediaFinal = calcularMedia(notas);
  print('Média: $mediaFinal');
}

double calcularMedia(List<double> notas) {
  if (notas.isEmpty) {
    return 0.0;
  }
  double soma = notas.reduce((valor, elemento) => valor + elemento);
  return soma / notas.length;
}
